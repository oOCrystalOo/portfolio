                def celsius_to_fahrenheit (celsius)
                    return (celsius * 9/5) + 32
                end

                puts "Enter degrees in celsius:"
                celsius = gets.chomp.to_f
                fahrenheit = celsius_to_fahrenheit(celsius)
                puts "#{celsius}°C is #{fahrenheit}°F."
            
        

        
                def calc_ordinal (num)
                    ones = num % 10
                    if num % 100 == 11 || num % 100 == 12 || num % 100 == 13
                        return "th"

                        if ones == 1
                            return "st"
                        elsif ones == 2
                            return "nd"
                        elsif ones == 3
                            return "rd"
                        else
                            return "th"
                        end
                    end
                end

                def foobar (num)
                    if num % 3 == 0 && num % 5 == 0
                        puts "Foobar"
                    elsif num % 3 == 0
                        puts "Foo"
                    elsif num % 5 == 0
                        puts "Bar"
                    else
                        puts num
                    end
                end

                function get_weekdays ($date_1, $date_2) {
                    /* $day_1, $day_2 formatted dates as yyyy-mm-dd
                    *  Example: day_1 = xxxx-xx-01, day_2 = xxxx-xx-17
                    *  [ S  M  T  W  T  F  S ] Days of week [0 - 6]
                    *    1 [2  3  4  5  6  7   [completed week] - Starts on the 2nd because we start counting on the day after
                    *    8][9  10 11 12 13 14  [completed week]
                    *    15]16 17              leftover days (2)
                    *  End result should be (5days * 2complete weeks) + 2 leftover week days = 12 weekdays
                    */
                    try {
                        $day_1 = new DateTime($date_1);
                        $day_2 = new DateTime($date_2);
                        $begin = $day_1;
                        $end = $day_2;
                        if ($day_1 > $day_2) {
                            $begin = $day_2;
                            $end = $day_1;
                        }
                        $diff = $day_1->diff($day_2);
                        $diff_days = $diff->days;                   //Get number of days between these 2 dates
                        $complete_weeks = floor($diff_days / 7);    //Get number of completed weeks.
                        $weekdays = $complete_weeks * 5;            //Each completed week will have 5 days
                        if ($diff_days % 7 !== 0) {
                            $leftover_days = $diff_days % 7;        //Get leftover days
                            $completed_days = 1 + ($complete_weeks * 7); //Start on 1 day after, and then add the bulk of completed weeks
                            $begin_date = $begin->modify('+' . $completed_days . ' day');
                            while ($leftover_days >= 0) {
                                $day_of_week = date_format($begin_date, 'w');
                                if ($day_of_week !== '0' && $day_of_week !== '6' ) {  //0,6 are weekend
                                    $weekdays++;
                                }
                                $begin_date->modify('+1 day');
                                $leftover_days--;
                            }
                        }
                        return $weekdays;
                    } catch (Exception $e) {
                        error_log(print_r($e, true));
                    }
                }
